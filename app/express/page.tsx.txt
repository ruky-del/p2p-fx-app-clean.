"use client";

import Link from "next/link";
import { useState } from "react";
import { FiHome, FiTrendingUp, FiUser } from "react-icons/fi";

export default function ExpressPage() {
  const [sendCurrency, setSendCurrency] = useState("GBP");
  const [receiveCurrency, setReceiveCurrency] = useState("TZS");
  const [sendAmount, setSendAmount] = useState("1000");

  const estimatedReceive =
    sendCurrency === "GBP" && receiveCurrency === "TZS"
      ? (Number(sendAmount) || 0) * 3120
      : sendCurrency === "TZS" && receiveCurrency === "GBP"
      ? (Number(sendAmount) || 0) / 3120
      : Number(sendAmount) || 0;

  const switchCurrencies = () => {
    setSendCurrency(receiveCurrency);
    setReceiveCurrency(sendCurrency);
  };

  const formatValue = (value: number, currency: string) => {
    if (currency === "GBP") {
      return `£${value.toLocaleString()}`;
    }
    return `${value.toLocaleString()} TZS`;
  };

  return (
    <main className="page">
      <div className="container">

        {/* HEADER */}
        <div className="market-header-card">
          <div className="market-topbar">
            <Link href="/" className="market-back">←</Link>

            <div className="market-tabs">
              <button className="market-tab market-tab-active">Express</button>
              <Link href="/market" className="market-tab">P2P</Link>
            </div>

            <div className="market-currency-pill">{sendCurrency}</div>
          </div>

          <h1 className="market-section-title">Express Exchange</h1>
          <p className="market-section-subtitle">
            Quick estimate between Tanzania and UK
          </p>

          <div className="market-filter-row">
            <select
              className="market-select"
              value={sendCurrency}
              onChange={(e) => setSendCurrency(e.target.value)}
            >
              <option value="GBP">GBP</option>
              <option value="TZS">TZS</option>
            </select>

            <select
              className="market-select"
              value={receiveCurrency}
              onChange={(e) => setReceiveCurrency(e.target.value)}
            >
              <option value="TZS">TZS</option>
              <option value="GBP">GBP</option>
            </select>

            <input
              className="market-input"
              type="number"
              placeholder="Amount"
              value={sendAmount}
              onChange={(e) => setSendAmount(e.target.value)}
            />

            <button className="btn btn-outline" onClick={switchCurrencies}>
              Swap
            </button>
          </div>
        </div>

        {/* RESULT */}
        <div className="market-section-card">
          <h2 className="market-section-title">Estimate</h2>

          <div className="market-offer-card">
            <div className="market-offer-left">

              <div className="market-detail-line">
                <span>You send</span>
                <strong>{formatValue(Number(sendAmount), sendCurrency)}</strong>
              </div>

              <div className="market-detail-line">
                <span>You receive</span>
                <strong>{formatValue(estimatedReceive, receiveCurrency)}</strong>
              </div>

            </div>
          </div>
        </div>

        {/* INFO */}
        <div className="card">
          <h2 className="card-title">Important</h2>
          <p className="card-subtitle">
            This is only an estimate. Final exchange happens via P2P traders.
          </p>
        </div>

        {/* NAV */}
        <div className="nav">
          <Link href="/">
            <FiHome />
            <span>Home</span>
          </Link>

          <Link href="/market">
            <FiTrendingUp />
            <span>Market</span>
          </Link>

          <Link href="/profile">
            <FiUser />
            <span>Profile</span>
          </Link>
        </div>

      </div>
    </main>
  );
}
